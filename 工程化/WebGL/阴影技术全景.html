一、Shadow Map 基础 ​

1.1 原理 ​

  1. 从光源视角渲染场景，把深度写入纹理
  2. 从相机视角渲染时，把像素转换到光源空间，比较深度
  3. 如果当前深度 > Shadow Map 深度 → 被遮挡 → 阴影
js
// Three.js 中使用
const light = new THREE.DirectionalLight(0xffffff, 1);
light.castShadow = true;
light.shadow.mapSize.width = 2048;
light.shadow.mapSize.height = 2048;
light.shadow.camera.near = 0.5;
light.shadow.camera.far = 50;
light.shadow.camera.left = -10;
light.shadow.camera.right = 10;
light.shadow.camera.top = 10;
light.shadow.camera.bottom = -10;

// 开启物体投射/接收阴影
mesh.castShadow = true;
mesh.receiveShadow = true;

1.2 PCF（百分比渐进过滤） ​

glsl
// 采样周围 3x3 像素做平均
float pcfShadow(sampler2D shadowMap, vec2 uv, float compare) {
    float result = 0.0;
    vec2 texelSize = 1.0 / vec2(textureSize(shadowMap, 0));

    for (int x = -1; x <= 1; x++) {
        for (int y = -1; y <= 1; y++) {
            vec2 offset = vec2(float(x), float(y)) * texelSize;
            float depth = texture2D(shadowMap, uv + offset).r;
            result += (compare > depth + bias) ? 0.0 : 1.0;
        }
    }
    return result / 9.0;
}

二、软阴影技术 ​

2.1 PCSS（Percentage Closer Soft Shadows） ​

根据遮挡物距离光源的距离动态调整采样半径：

glsl
float pcssShadow(sampler2D shadowMap, vec3 shadowCoord, float lightSize) {
    // 1. Blocker Search：找平均遮挡深度
    float blockerDepth = 0.0;
    int blockers = 0;
    float searchRadius = 0.05;

    for (int i = 0; i < 16; i++) {
        vec2 offset = poissonDisk[i] * searchRadius;
        float depth = texture2D(shadowMap, shadowCoord.xy + offset).r;
        if (depth < shadowCoord.z) {
            blockerDepth += depth;
            blockers++;
        }
    }

    if (blockers == 0) return 1.0;  // 无遮挡
    if (blockers == 16) return 0.0; // 完全遮挡

    blockerDepth /= float(blockers);

    // 2. 计算半影半径
    float penumbra = (shadowCoord.z - blockerDepth) * lightSize / blockerDepth;

    // 3. PCF 过滤
    return pcfShadow(shadowMap, shadowCoord.xy, shadowCoord.z, penumbra);
}

2.2 VSM（Variance Shadow Maps） ​

存储深度和深度²，用切比雪夫不等式估计遮挡概率：

glsl
// VSM 生成 pass
void main() {
    float depth = gl_FragCoord.z;
    gl_FragColor = vec4(depth, depth * depth, 0.0, 1.0);
}

// VSM 采样
float vsmShadow(sampler2D shadowMap, vec2 uv, float compare) {
    vec2 moments = texture2D(shadowMap, uv).rg;

    if (compare <= moments.x) return 1.0;

    float variance = moments.y - moments.x * moments.x;
    variance = max(variance, 0.00002);

    float d = compare - moments.x;
    float pMax = variance / (variance + d * d);

    return smoothstep(0.2, 1.0, pMax);
}

优点： 无需过滤即可得到软阴影，支持预过滤 缺点： Light leaking（漏光），需要修正

2.3 CSM（Cascaded Shadow Maps） ​

把视锥体分割为多层，每层用不同分辨率的 Shadow Map：

js
const csm = new CSM({
    maxFar: 100,
    cascades: 4,
    shadowMapSize: 2048,
    lightDirection: new THREE.Vector3(1, -1, 1),
    camera: camera
});

csm.setupMaterial(material);  // 自动注入 shader 代码
层级覆盖范围分辨率利用率
Cascade 0近处 0-10m高
Cascade 110-30m中
Cascade 230-60m低
Cascade 360-100m很低

三、接触阴影（Contact Shadows） ​

3.1 屏幕空间阴影 ​

对近距离阴影用屏幕空间 Ray Marching：

glsl
float contactShadow(vec3 worldPos, vec3 lightDir) {
    vec3 ro = worldPos + normal * 0.01;
    vec3 rd = lightDir;

    for (int i = 0; i < 32; i++) {
        float t = 0.1 + float(i) * 0.05;
        vec3 p = ro + rd * t;
        vec4 screenPos = projectionMatrix * viewMatrix * vec4(p, 1.0);
        vec2 uv = screenPos.xy / screenPos.w * 0.5 + 0.5;
        float depth = texture2D(uDepthTexture, uv).r;
        float sceneDepth = getLinearDepth(depth);
        float sampleDepth = getLinearDepth(screenPos.z / screenPos.w);

        if (sceneDepth < sampleDepth - 0.01) return 0.0;
    }
    return 1.0;
}

四、Ray Traced Shadows ​

glsl
// 在 ray marching 场景中做真·阴影
float shadowRay(vec3 ro, vec3 rd, float maxDist) {
    float t = 0.01;
    for (int i = 0; i < 64; i++) {
        float d = map(ro + rd * t);
        if (d < 0.001) return 0.0;
        t += d;
        if (t > maxDist) break;
    }
    return 1.0;
}

glsl
// Slope-scaled depth bias
float bias = max(0.05 * (1.0 - dot(normal, lightDir)), 0.005);

// 阴影远离光源时淡化
