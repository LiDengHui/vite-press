一、光线的数学定义 ​

1.1 参数方程 ​

光线由起点 ro (ray origin) 和单位方向 rd (ray direction) 定义：

glsl
// 光线上的点：P(t) = ro + t * rd,  t ∈ [t_min, t_max]
// t > 0 表示前方，t < 0 表示后方（通常忽略）
vec3 rayAt(float t) {
    return ro + t * rd;
}

1.2 光线生成（从相机） ​

glsl
// 从相机位置生成指向像素的光线
vec3 getRay(vec2 pixel, vec2 resolution, vec3 camPos, vec3 camDir, float fov) {
    vec2 uv = (pixel - 0.5 * resolution) / resolution.y;
    float focalLength = 1.0 / tan(fov * 0.5);

    // 构建相机坐标系
    vec3 forward = normalize(camDir);
    vec3 right = normalize(cross(forward, vec3(0, 1, 0)));
    vec3 up = cross(right, forward);

    vec3 rd = normalize(uv.x * right + uv.y * up + focalLength * forward);
    return rd;
}

二、光线与几何体相交 ​

2.1 球体相交（解析法） ​

球面方程：|P - C|² = r²

代入光线方程 |ro + t*rd - C|² = r²，令 oc = ro - C：

|rd|² t² + 2(oc·rd) t + |oc|² - r² = 0

因为 rd 是单位向量，|rd|² = 1：

glsl
bool intersectSphere(vec3 ro, vec3 rd, vec3 center, float radius, out float t) {
    vec3 oc = ro - center;
    float b = dot(oc, rd);           // 原式是 2b，这里用 b 代替
    float c = dot(oc, oc) - radius * radius;
    float discriminant = b * b - c;   // 简化后的判别式（a=1）

    if (discriminant < 0.0) return false;

    float sqrtD = sqrt(discriminant);
    float t1 = -b - sqrtD;   // 近交点
    float t2 = -b + sqrtD;   // 远交点

    if (t1 > 0.001) { t = t1; return true; }  // 0.001 避免自相交
    if (t2 > 0.001) { t = t2; return true; }
    return false;  // 两个交点都在后方
}

2.2 三角形相交（Möller–Trumbore 算法） ​

glsl
// 光线与三角形相交（Möller-Trumbore）
bool intersectTriangle(
    vec3 ro, vec3 rd,
    vec3 v0, vec3 v1, vec3 v2,
    out float t, out vec3 outNormal
) {
    vec3 edge1 = v1 - v0;
    vec3 edge2 = v2 - v0;
    vec3 h = cross(rd, edge2);
    float a = dot(edge1, h);

    if (abs(a) < 0.00001) return false;  // 光线与三角形平行

    float f = 1.0 / a;
    vec3 s = ro - v0;
    float u = f * dot(s, h);
    if (u < 0.0 || u > 1.0) return false;

    vec3 q = cross(s, edge1);
    float v = f * dot(rd, q);
    if (v < 0.0 || u + v > 1.0) return false;

    t = f * dot(edge2, q);
    if (t > 0.00001) {
        outNormal = normalize(cross(edge1, edge2));
        return true;
    }
    return false;
}

2.3 AABB（轴对齐包围盒）相交 ​

glsl
bool intersectAABB(vec3 ro, vec3 rd, vec3 boxMin, vec3 boxMax, out float tMin, out float tMax) {
    vec3 invD = 1.0 / rd;
    vec3 t0 = (boxMin - ro) * invD;
    vec3 t1 = (boxMax - ro) * invD;

    vec3 tNear = min(t0, t1);
    vec3 tFar = max(t0, t1);

    tMin = max(max(tNear.x, tNear.y), tNear.z);
    tMax = min(min(tFar.x, tFar.y), tFar.z);

    return tMin <= tMax && tMax > 0.0;
}

2.4 平面相交 ​

glsl
// 平面：n·p + d = 0（n 是单位法线）
float intersectPlane(vec3 ro, vec3 rd, vec3 n, float d) {
    float denom = dot(n, rd);
    if (abs(denom) < 0.0001) return -1.0;
    return -(dot(n, ro) + d) / denom;
}

三、反射与折射 ​

3.1 反射定律 ​

入射向量 I（指向表面）、法线 N，反射方向 R：

glsl
vec3 reflect(vec3 I, vec3 N) {
    // R = I - 2(I·N)N
    return I - 2.0 * dot(I, N) * N;
}

3.2 斯涅尔折射定律 ​

glsl
// η = η₁/η₂（入射介质折射率 / 折射介质折射率）
// 空气→水 ≈ 1.0/1.33，空气→玻璃 ≈ 1.0/1.5
vec3 refract(vec3 I, vec3 N, float eta) {
    float cosi = dot(-I, N);  // 入射角余弦
    float cost2 = 1.0 - eta * eta * (1.0 - cosi * cosi);

    if (cost2 < 0.0) {
        // 全内反射
        return reflect(I, N);
    }

    return eta * I + (eta * cosi - sqrt(cost2)) * N;
}

3.3 菲涅尔方程（Schlick 近似） ​

反射率随视角变化：正看时反射少，斜看时反射多（水面效应）。

glsl
float fresnelSchlick(float cosTheta, float F0) {
    // F0 = ((n1 - n2)/(n1 + n2))² 是垂直入射时的反射率
    return F0 + (1.0 - F0) * pow(1.0 - cosTheta, 5.0);
}

// 完整版：考虑折射率
float fresnel(float cosi, float etai, float etat) {
    float sint = etai / etat * sqrt(max(0.0, 1.0 - cosi * cosi));
    if (sint >= 1.0) return 1.0;  // 全反射

    float cost = sqrt(max(0.0, 1.0 - sint * sint));
    cosi = abs(cosi);

    float Rs = ((etat * cosi) - (etai * cost)) / ((etat * cosi) + (etai * cost));
    float Rp = ((etai * cosi) - (etat * cost)) / ((etai * cosi) + (etat * cost));
    return (Rs * Rs + Rp * Rp) / 2.0;
}

四、光照模型 ​

4.1 Lambert 漫反射 ​

glsl
float lambert(vec3 N, vec3 L) {
    return max(dot(N, L), 0.0);
}

4.2 Phong 光照模型（1975） ​

glsl
vec3 phong(
    vec3 N, vec3 L, vec3 V,   // 法线、光方向、视线方向（都需单位化）
    vec3 diffuseColor, vec3 specularColor,
    float shininess
) {
    vec3 R = reflect(-L, N);  // 完美反射方向
    float diff = max(dot(N, L), 0.0);
    float spec = pow(max(dot(R, V), 0.0), shininess);

    return diffuseColor * diff + specularColor * spec;
}

Phong 的问题： 当视线和入射光在法线两侧时，dot(R, V) 可能为负导致 specular 消失（边缘过暗）。

4.3 Blinn-Phong（1977） ​

用半角向量 H = (L + V) / |L + V| 代替反射向量：

glsl
vec3 blinnPhong(
    vec3 N, vec3 L, vec3 V,
    vec3 diffuseColor, vec3 specularColor,
    float shininess
) {
    vec3 H = normalize(L + V);
    float diff = max(dot(N, L), 0.0);
    float spec = pow(max(dot(N, H), 0.0), shininess * 4.0);  // 指数通常乘 4

    return diffuseColor * diff + specularColor * spec;
}

优点：

  • 没有 dot(R,V) < 0 的问题
  • 计算更快（不用 reflect）
  • 更符合物理直觉

4.4 Gouraud vs Phong 着色 ​

方法计算位置效果性能
Gouraud顶点着色器插值颜色，高光会变形快
Phong片段着色器逐像素光照，正确慢
glsl
// Gouraud（顶点着色器计算颜色，varying 插值）
varying vec3 vColor;
void main() {
    vColor = phongLighting(...);  // 在顶点算
}

// Phong（片段着色器计算，varying 传法线）
varying vec3 vNormal, vLightDir, vViewDir;
void main() {
    gl_FragColor = vec4(phongLighting(vNormal, vLightDir, vViewDir, ...), 1.0);
}

4.5 阴影计算（硬阴影） ​

glsl
// 从碰撞点朝光源发射 shadow ray
float calcShadow(vec3 hitPoint, vec3 lightPos) {
    vec3 shadowRo = hitPoint + normal * 0.001;  // 偏移避免自相交
    vec3 shadowRd = normalize(lightPos - hitPoint);
    float lightDist = length(lightPos - hitPoint);

    float t = rayMarch(shadowRo, shadowRd);  // 或解析求交
    return (t < lightDist) ? 0.0 : 1.0;      // 被遮挡 = 阴影
}

4.6 软阴影（Penumbra） ​

glsl
// 通过光源半径计算半影
float calcSoftShadow(vec3 ro, vec3 rd, float mint, float maxt, float k) {
    float res = 1.0;
    float t = mint;
    for (int i = 0; i < 50; i++) {
        float h = map(ro + rd * t);  // SDF 距离
        if (h < 0.001) return 0.0;
        res = min(res, k * h / t);
        t += clamp(h, 0.01, 0.5);
        if (t > maxt) break;
    }
    return clamp(res, 0.0, 1.0);
}
// k 越大阴影越锐利（100 = 硬阴影，8 = 软阴影）

五、PBR 基础（Cook-Torrance） ​

现代物理渲染的核心是微表面模型。

glsl
// 简化版 Cook-Torrance BRDF
vec3 pbrLighting(
    vec3 N, vec3 V, vec3 L,
    vec3 albedo, float metallic, float roughness
) {
    vec3 H = normalize(L + V);
    float NdotL = max(dot(N, L), 0.0);
    float NdotV = max(dot(N, V), 0.0);
    float NdotH = max(dot(N, H), 0.0);
    float VdotH = max(dot(V, H), 0.0);

    // 法线分布函数 (Trowbridge-Reitz GGX)
    float alpha = roughness * roughness;
    float alpha2 = alpha * alpha;
    float denom = NdotH * NdotH * (alpha2 - 1.0) + 1.0;
    float D = alpha2 / (3.14159 * denom * denom);

    // 几何遮蔽 (Schlick-GGX)
    float k = alpha / 2.0;
    float G1L = NdotL / (NdotL * (1.0 - k) + k);
    float G1V = NdotV / (NdotV * (1.0 - k) + k);
    float G = G1L * G1V;

    // Fresnel (Schlick)
    vec3 F0 = mix(vec3(0.04), albedo, metallic);  // 非金属 F0 ≈ 0.04
    vec3 F = F0 + (1.0 - F0) * pow(1.0 - VdotH, 5.0);

    // 镜面反射项
    vec3 specular = (D * G * F) / (4.0 * NdotL * NdotV + 0.001);

    // 漫反射（金属没有漫反射）
    vec3 diffuse = (1.0 - metallic) * albedo / 3.14159;

    return (diffuse + specular) * NdotL;
}

六、完整示例：光线运算 + Phong 着色 ​

js
import * as THREE from 'three';

const vertexShader = `
    varying vec3 vWorldPos;
    varying vec3 vNormal;
    void main() {
        vWorldPos = (modelMatrix * vec4(position, 1.0)).xyz;
        vNormal = normalize(normalMatrix * normal);
        gl_Position = projectionMatrix * viewMatrix * modelMatrix * vec4(position, 1.0);
    }
`;

const fragmentShader = `
    precision highp float;
    varying vec3 vWorldPos;
    varying vec3 vNormal;

    uniform vec3 uLightPos[3];
    uniform vec3 uLightColor[3];
    uniform vec3 uCameraPos;

    vec3 phong(vec3 N, vec3 V, vec3 L, vec3 lightColor, vec3 diff, vec3 spec, float shine) {
        vec3 R = reflect(-L, N);
        float diffTerm = max(dot(N, L), 0.0);
        float specTerm = pow(max(dot(R, V), 0.0), shine);
        return lightColor * (diff * diffTerm + spec * specTerm);
    }

    void main() {
        vec3 N = normalize(vNormal);
        vec3 V = normalize(uCameraPos - vWorldPos);
        vec3 color = vec3(0.05);  // 环境光

        for (int i = 0; i < 3; i++) {
            vec3 L = normalize(uLightPos[i] - vWorldPos);
            color += phong(N, V, L, uLightColor[i], vec3(0.8, 0.3, 0.2), vec3(1.0), 32.0);
        }

        gl_FragColor = vec4(color, 1.0);
    }
`;
