Electron зғӯжӣҙж–° вҖӢ

Electron еә”з”Ёе®һзҺ°еңЁзәҝжӣҙж–°пјҲд№ҹз§°дёәзғӯжӣҙж–°пјүпјҢиғҪи®©дҪ зҡ„еә”з”ЁеңЁдёҚйңҖиҰҒз”ЁжҲ·йҮҚж–°дёӢиҪҪе®Ңж•ҙе®үиЈ…еҢ…зҡ„жғ…еҶөдёӢпјҢиҺ·еҸ–ж–°еҠҹиғҪжҲ–дҝ®еӨҚй—®йўҳпјҢжҳҫи‘—жҸҗеҚҮз”ЁжҲ·дҪ“йӘҢгҖӮзӣ®еүҚдё»жөҒзҡ„е®һзҺ°ж–№ејҸжңү дҪҝз”Ё electron-updater жЁЎеқ—гҖҒдҫқжүҳ Electron еҶ…зҪ®зҡ„ autoUpdater жЁЎеқ—зӯүж–№жЎҲгҖӮ

дёӢйқўжҳҜдёҖдёӘеҜ№жҜ”иЎЁж јпјҢеё®еҠ©дҪ еҝ«йҖҹдәҶи§ЈеҮ з§Қдё»иҰҒж–№ејҸзҡ„зү№зӮ№пјҡ

жӣҙж–°ж–№ејҸдјҳзӮ№зјәзӮ№йҖӮз”ЁеңәжҷҜ
electron-updaterеҠҹиғҪдё°еҜҢпјҢж”ҜжҢҒеўһйҮҸжӣҙж–°пјҢдёҺ electron-builder йӣҶжҲҗеәҰй«ҳйңҖиҰҒй…ҚзҪ®жү“еҢ…е·Ҙе…· (еҰӮ electron-builder)еӨ§еӨҡж•°е•ҶдёҡйЎ№зӣ®пјҢе°Өе…¶жҳҜйңҖиҰҒзЁіе®ҡгҖҒеҸҜйқ жӣҙж–°жөҒзЁӢе’ҢеҸҜиғҪж¶үеҸҠеӨ§йҮҸж–Үд»¶еҸҳеҠЁзҡ„йЎ№зӣ®
еҶ…зҪ® autoUpdaterElectron еҺҹз”ҹж”ҜжҢҒпјҢж— йңҖйўқеӨ–е®үиЈ…дёҚеҗҢе№іеҸ°иЎҢдёәдёҚдёҖпјҲmacOS йңҖзӯҫеҗҚпјүпјҢйңҖиҮӘе»әжӣҙж–°жңҚеҠЎеҷЁе№¶жҸҗдҫӣзү№е®ҡж јејҸзҡ„жӣҙж–°дҝЎжҒҜеҜ№жӣҙж–°жөҒзЁӢжңүй«ҳеәҰе®ҡеҲ¶йңҖжұӮпјҢж„ҝж„ҸиҮӘиЎҢеӨ„зҗҶеҗ„е№іеҸ°е·®ејӮе’ҢжңҚеҠЎеҷЁйҖ»иҫ‘зҡ„йЎ№зӣ®
жүӢеҠЁе®һзҺ°зғӯжӣҙж–°жһҒеәҰзҒөжҙ»пјҢеҸҜжҺ§жҖ§ејәпјҢеҸҜеҸӘжӣҙж–°жёІжҹ“иҝӣзЁӢзӣёе…іж–Үд»¶е®һзҺ°еӨҚжқӮпјҢйңҖиҮӘиЎҢеӨ„зҗҶзүҲжң¬еҜ№жҜ”гҖҒдёӢиҪҪгҖҒжӣҝжҚўгҖҒж ЎйӘҢзӯүжүҖжңүзҺҜиҠӮпјҢеҸҜйқ жҖ§йңҖиҮӘиЎҢдҝқйҡңжӣҙж–°еҶ…е®№жһҒе°‘пјҲеҰӮд»…дҝ®ж”№ HTML/CSS/JSпјүгҖҒеёҢжңӣж·ұеәҰе®ҡеҲ¶жӣҙж–°йҖ»иҫ‘жҲ–еӯҰд№ з ”з©¶
update-electron-appи®ҫзҪ®з®ҖеҚ•пјҢж— йңҖиҮӘе»әжңҚеҠЎеҷЁпјҲдҪҝз”Ё Electron е®ҳж–№жңҚеҠЎпјүпјҢйҖӮеҗҲејҖжәҗйЎ№зӣ®д»…йҖӮз”ЁдәҺе…¬ејҖзҡ„ GitHub ReleasesпјҢеҠҹиғҪзӣёеҜ№з®ҖеҚ•пјҢеҸҜиғҪдёҚж»Ўи¶іеӨҚжқӮйңҖжұӮејҖжәҗйЎ№зӣ®пјҢдё”зүҲжң¬еҸ‘еёғйҖҡиҝҮ GitHub Releases иҝӣиЎҢ

в„№пёҸ жіЁж„Ҹпјҡж— и®әйҮҮз”Ёе“Әз§Қж–№ејҸпјҢеә”з”ЁзӯҫеҗҚе’ҢзүҲжң¬з®ЎзҗҶйғҪиҮіе…ійҮҚиҰҒгҖӮеә”з”Ёеҝ…йЎ»жңүжңүж•ҲзӯҫеҗҚпјҢеҗҰеҲҷжӣҙж–°еҗҺеҸҜиғҪж— жі•иҝҗиЎҢжҲ–иў«жқҖжҜ’иҪҜд»¶иҜҜжҠҘгҖӮзүҲжң¬еҸ·еә”йҒөеҫӘиҜӯд№үеҢ–зүҲжң¬жҺ§еҲ¶пјҲSemantic VersioningпјүпјҢд»ҘдҫҝжӯЈзЎ®еҲӨж–ӯжҳҜеҗҰйңҖиҰҒжӣҙж–°гҖӮ

рҹ”§ ж–№жЎҲдёҖпјҡдҪҝз”Ё electron-updaterпјҲжҺЁиҚҗпјү вҖӢ

electron-updater жҳҜ electron-builder еҶ…зҪ®зҡ„жЁЎеқ—пјҢеҠҹиғҪејәеӨ§пјҢж”ҜжҢҒ**иҮӘеҠЁе·®еҲҶжӣҙж–° **пјҲdelta updatesпјүпјҢеҸӘйңҖдёӢиҪҪеҸҳжӣҙйғЁеҲҶпјҢиҠӮзңҒз”ЁжҲ·жөҒйҮҸе’Ңж—¶й—ҙгҖӮ

    ipcRenderer.send('check_for_updates'); // дё»иҝӣзЁӢйңҖзӣ‘еҗ¬жӯӨдәӢд»¶е№¶и°ғз”Ё autoUpdater.checkForUpdates()
});

  4. жү“еҢ…дёҺеҸ‘еёғ еңЁ package.json дёӯй…ҚзҪ® electron-builder зҡ„ publish еӯ—ж®өпјҢе‘ҠзҹҘжү“еҢ…е·Ҙе…·еҰӮдҪ•еҸ‘еёғзүҲжң¬е’Ңз”ҹжҲҗжӣҙж–°дҝЎжҒҜпјҡ

    json
    {
    "build": {
        "appId": "com.yourcompany.yourapp",
        "productName": "YourApp",
        "directories": {
            "output": "dist"
        },
        "publish": [
            {
                "provider": "generic", // д№ҹеҸҜд»ҘжҳҜ 'github', 's3' зӯү
                "url": "https://your-update-server.com/updates/" // жӣҙж–°ж–Үд»¶еӯҳж”ҫзҡ„еҹәзЎҖ URL
            }
        ]
    }
}

    дҪҝз”Ё electron-builder жү“еҢ…ж—¶пјҢе®ғдјҡз”ҹжҲҗ latest.ymlпјҲжҲ– latest-mac.yml зӯүпјүгҖҒblockmap зӯүж–Үд»¶гҖӮдҪ йңҖиҰҒ* *е°Ҷиҝҷдәӣж–Үд»¶иҝһеҗҢжү“еҢ…з”ҹжҲҗзҡ„е®үиЈ…зЁӢеәҸпјҲеҰӮ .exe, .dmg, .AppImageпјүдёҖиө·дёҠдј еҲ°дҪ зҡ„жӣҙж–°жңҚеҠЎеҷЁ**пјҲpublish дёӯ url жүҖжҢҮеҗ‘зҡ„ең°еқҖпјүгҖӮ

рҹҢҗ ж–№жЎҲдәҢпјҡдҪҝз”Ё Electron еҶ…зҪ®зҡ„ autoUpdater жЁЎеқ— вҖӢ

Electron жң¬иә«жҸҗдҫӣдәҶ autoUpdater жЁЎеқ—пјҢдҪҶе…¶жң¬иә«* *еҸӘиҙҹиҙЈжЈҖжҹҘжӣҙж–°гҖҒдёӢиҪҪе’Ңе®үиЈ…пјҢйңҖиҰҒдҪ иҮӘиЎҢжҗӯе»әжӣҙж–°жңҚеҠЎеҷЁ**е№¶жҸҗдҫӣзү№е®ҡж јејҸзҡ„жӣҙж–°дҝЎжҒҜгҖӮ

  1. дё»иҝӣзЁӢй…ҚзҪ®

    javascript
    const { autoUpdater } = require('electron'); // жіЁж„ҸпјҡиҝҷйҮҢжҳҜ Electron иҮӘеёҰзҡ„жЁЎеқ—

// и®ҫзҪ®жӣҙж–°жңҚеҠЎеҷЁзҡ„ URLпјҢURL еә”жҢҮеҗ‘дёҖдёӘжҸҗдҫӣжӣҙж–°дҝЎжҒҜзҡ„ JSON ж–Үд»¶пјҲеҰӮ updates.jsonпјү
autoUpdater.setFeedURL({
    url: 'https://your-update-server.com/updates?version=' + app.getVersion() + '&platform=' + process.platform
});

// жЈҖжҹҘжӣҙж–°
autoUpdater.checkForUpdates();

// зӣ‘еҗ¬дәӢд»¶пјҲдёҺ electron-updater зұ»дјјпјү
autoUpdater.on('update-available', () => {
    console.log('Update available.');
});
autoUpdater.on('update-downloaded', (event, releaseNotes, releaseName, releaseDate, updateUrl, quitAndUpdate) => {
    console.log('Update downloaded.');
    // жҸҗзӨәз”ЁжҲ·йҮҚеҗҜ
    autoUpdater.quitAndInstall(); // жҲ–иҖ…и®©з”ЁжҲ·йҖүжӢ©ж—¶жңә
});
autoUpdater.on('error', (error) => {
    console.error('Error in auto-updater.', error);
});

  2. жҗӯе»әжӣҙж–°жңҚеҠЎеҷЁ дҪ йңҖиҰҒдёҖдёӘжңҚеҠЎеҷЁпјҢж №жҚ®е®ўжҲ·з«ҜиҜ·жұӮпјҲйҖҡеёёеҢ…еҗ«еҪ“еүҚзүҲжң¬е’Ңе№іеҸ°пјүиҝ”еӣһдёҖдёӘ Squirrel е…је®№зҡ„ JSON е“Қеә”гҖӮдҫӢеҰӮпјҢеҜ№дәҺ Windows (url еҸӮж•°дёӯ platform=win32)пјҢиҝ”еӣһеҶ…е®№еҸҜиғҪзұ»дјјпјҡ

    json
    {
    "url": "https://your-update-server.com/releases/YourApp-1.2.0-full.nupkg",
    "name": "1.2.0",
    "notes": "дҝ®еӨҚдәҶиӢҘе№Ій—®йўҳпјҢеўһеҠ дәҶж–°еҠҹиғҪ",
    "pub_date": "2023-04-01T12:00:00.000Z"
}

    жңҚеҠЎеҷЁйңҖиҰҒж №жҚ®иҜ·жұӮзҡ„зүҲжң¬еҸ·еҲӨж–ӯжҳҜеҗҰжңүжӣҙж–°пјҢе№¶иҝ”еӣһеҜ№еә”ж–°зүҲе®үиЈ…еҢ…пјҲеҰӮ .nupkg for Windows, .zip for macOSпјүзҡ„дҝЎжҒҜгҖӮmacOS иҝҳйңҖеӨ„зҗҶ RELEASES ж–Үд»¶гҖӮ

рҹ“қ ж–№жЎҲдёүпјҡжүӢеҠЁе®һзҺ°зғӯжӣҙж–°пјҲйҖӮз”ЁдәҺжёІжҹ“иҝӣзЁӢпјү вҖӢ

еҰӮжһңдҪ зҡ„жӣҙж–°еҸӘж¶үеҸҠжёІжҹ“иҝӣзЁӢзҡ„д»Јз ҒпјҲHTML, CSS, JSпјүдё”дё»иҝӣзЁӢд»Јз ҒжңӘеҸҳеҢ–пјҢеҸҜд»ҘиҖғиҷ‘жӣҙиҪ»йҮҸзҡ„жүӢеҠЁзғӯжӣҙж–°гҖӮ

  1. еҺҹзҗҶпјҡеә”з”ЁеҗҜеҠЁж—¶жҲ–е®ҡжңҹд»ҺиҝңзЁӢжңҚеҠЎеҷЁпјҲеҰӮ GitHub Raw ContentгҖҒиҮӘе»ә CDNпјүжЈҖжҹҘдёҖдёӘзүҲжң¬жҸҸиҝ°ж–Үд»¶пјҲеҰӮ package.json жҲ–иҮӘе®ҡд№үзҡ„ version.jsonпјүгҖӮжҜ”иҫғиҝңзЁӢзүҲжң¬дёҺжң¬ең°зүҲжң¬гҖӮеҰӮжһңиҝңзЁӢзүҲжң¬жӣҙж–°пјҢеҲҷйҖҗдёӘдёӢиҪҪжңүеҸҳеҠЁзҡ„жёІжҹ“иҝӣзЁӢж–Үд»¶пјҲжҲ–дёҖдёӘеўһйҮҸеҢ…пјүеҲ°дёҙж—¶зӣ®еҪ•пјҢ йӘҢиҜҒеҗҺиҰҶзӣ–жң¬ең°ж–Үд»¶пјҢжңҖеҗҺеҲ·ж–°йЎөйқўжҲ–жҸҗзӨәз”ЁжҲ·йҮҚеҗҜжёІжҹ“иҝӣзЁӢгҖӮ

  2. з®ҖеҚ•зӨәдҫӢпјҡ

    javascript
    // еңЁжёІжҹ“иҝӣзЁӢжҲ–дё»иҝӣзЁӢдёӯпјҲйңҖиҖғиҷ‘е®үе…ЁжҖ§е’Ңжқғйҷҗпјү
const https = require('https');
const fs = require('fs');
const path = require('path');

function checkForUpdates() {
    // 1. иҺ·еҸ–иҝңзЁӢзүҲжң¬дҝЎжҒҜ
    const remoteVersionUrl = 'https://raw.githubusercontent.com/yourusername/yourrepo/master/version.json';
    https
        .get(remoteVersionUrl + '?t=' + Date.now(), (res) => {
            // еҠ ж—¶й—ҙжҲійҒҝе…Қзј“еӯҳ
            let data = '';
            res.on('data', (chunk) => (data += chunk));
            res.on('end', () => {
                const remoteInfo = JSON.parse(data);
                const localVersion = require('./package.json').version; // жң¬ең°зүҲжң¬

                // 2. жҜ”иҫғзүҲжң¬
                if (isNewerVersion(remoteInfo.version, localVersion)) {
                    // 3. дёӢиҪҪжӣҙж–°ж–Үд»¶еҲ—иЎЁдёӯзҡ„ж–Үд»¶
                    downloadUpdateFiles(remoteInfo.files); // remoteInfo.files еҸҜиғҪжҳҜдёҖдёӘйңҖиҰҒжӣҙж–°зҡ„ж–Үд»¶еҲ—иЎЁ
                }
            });
        })
        .on('error', (err) => {
            console.error('Failed to check for updates:', err);
        });
}

function isNewerVersion(remote, local) {
    // з®ҖеҚ•зҡ„зүҲжң¬жҜ”иҫғйҖ»иҫ‘пјҢеҸҜдҪҝз”Ё semver еә“жӣҙдёҘи°Ё
    const remoteParts = remote.split('.').map(Number);
    const localParts = local.split('.').map(Number);
    for (let i = 0; i < Math.max(remoteParts.length, localParts.length); i++) {
        const r = remoteParts[i] || 0;
        const l = localParts[i] || 0;
        if (r > l) return true;
        if (r < l) return false;
    }
    return false;
}

function downloadUpdateFiles(filesToUpdate) {
    filesToUpdate.forEach((filePath) => {
        const remoteFileUrl = `https://raw.githubusercontent.com/yourusername/yourrepo/master/${filePath}`;
        const localFilePath = path.join(__dirname, filePath);
        // еҲӣе»әзӣ®еҪ•пјҲеҰӮжһңдёҚеӯҳеңЁпјү
        fs.mkdirSync(path.dirname(localFilePath), { recursive: true });

        https
            .get(remoteFileUrl, (res) => {
                let fileData = '';
                res.setEncoding('utf8');
                res.on('data', (chunk) => (fileData += chunk));
                res.on('end', () => {
                    // е°Ҷжӣҙж–°ж–Үд»¶еҶҷе…Ҙдёҙж—¶дҪҚзҪ®пјҢе…ЁйғЁжҲҗеҠҹеҗҺиҝӣиЎҢжӣҝжҚўпјҲжңҖеҘҪжңүж ЎйӘҢпјҢеҰӮе“ҲеёҢпјү
                    const tempFilePath = localFilePath + '.tmp';
                    fs.writeFileSync(tempFilePath, fileData);
                    // йӘҢиҜҒж–Үд»¶е®Ңж•ҙжҖ§пјҲеҸҜйҖүдҪҶжҺЁиҚҗпјү
                    // if (validateFile(tempFilePath)) {
                    fs.renameSync(tempFilePath, localFilePath); // еҺҹеӯҗжӣҝжҚў
                    console.log(`Updated: ${filePath}`);
                    // жүҖжңүж–Үд»¶жӣҙж–°е®ҢжҲҗеҗҺпјҢжҸҗзӨәз”ЁжҲ·еҲ·ж–°жҲ–еә”з”Ёе·ІиҮӘеҠЁжӣҙж–°
                    // }
                });
            })
            .on('error', (err) => {
                console.error(`Failed to download ${filePath}:`, err);
            });
    });
}

// еә”з”ЁеҗҜеҠЁж—¶жЈҖжҹҘ
checkForUpdates();

    вҡ пёҸ жіЁж„ҸпјҡжӯӨж–№жі•е®үе…ЁжҖ§иҫғдҪҺпјҢйңҖиҰҒи°Ёж…ҺеӨ„зҗҶж–Үд»¶дёӢиҪҪе’ҢиҰҶзӣ–пјҢйҒҝе…Қе®үе…ЁжјҸжҙһгҖӮд»…йҖӮз”ЁдәҺз®ҖеҚ•еңәжҷҜгҖӮ

рҹ’Һ ж–№жЎҲеӣӣпјҡдҪҝз”Ё update-electron-appпјҲй’ҲеҜ№ејҖжәҗйЎ№зӣ®пјү вҖӢ

еҰӮжһңдҪ зҡ„ Electron еә”з”ЁжҳҜејҖжәҗйЎ№зӣ®пјҢе№¶дё”дҪҝз”Ё GitHub Releases еҸ‘еёғзүҲжң¬пјҢеҸҜд»ҘиҖғиҷ‘ Electron е®ҳж–№жҸҗдҫӣзҡ„ update-electron-app жЁЎеқ—гҖӮ

  1. е®үиЈ…пјҡ

    bash
    npm install update-electron-app

  2. дҪҝз”Ёпјҡ еңЁдё»иҝӣзЁӢзҡ„д»»дҪ•ең°ж–№и°ғз”Ёпјҡ

    javascript
    require('update-electron-app')();

    иҜҘжЁЎеқ—дјҡеңЁеә”з”ЁеҗҜеҠЁж—¶е’Ңд№ӢеҗҺжҜҸйҡ”дёҖж®өж—¶й—ҙпјҲеҰӮ10еҲҶй’ҹпјүиҮӘеҠЁжЈҖжҹҘжӣҙж–°гҖӮе®ғдҪҝз”Ё GitHub Releases дёӯзҡ„дҝЎжҒҜпјҢеҸ‘зҺ°жӣҙж–°еҗҺдјҡиҮӘеҠЁдёӢиҪҪе№¶еңЁе°ұз»Әж—¶жҸҗзӨәз”ЁжҲ·гҖӮ

вҡ пёҸ жіЁж„ҸдәӢйЎ№дёҺеёёи§Ғй—®йўҳ вҖӢ

  1. д»Јз ҒзӯҫеҗҚпјҲCode SigningпјүпјҡиҮіе…ійҮҚиҰҒпјҒж— и®әжҳҜ macOS иҝҳжҳҜ WindowsпјҢжІЎжңүжӯЈзЎ®зӯҫеҗҚзҡ„еә”з”Ёе°Ҷж— жі•жӯЈеёёиҝӣиЎҢиҮӘеҠЁжӣҙж–° пјҢз”ҡиҮіеҸҜиғҪж— жі•иҝҗиЎҢгҖӮmacOS йҖҡеёёйңҖиҰҒ Apple Developer иҜҒд№ҰпјҢWindows еҲҷеҸҜд»ҘдҪҝз”Ёд»Јз ҒзӯҫеҗҚиҜҒд№ҰпјҲеҰӮжқҘиҮӘ DigiCert, Sectigo зӯүпјүгҖӮ
  2. жӣҙж–°жңҚеҠЎеҷЁй…ҚзҪ®пјҡзЎ®дҝқдҪ зҡ„жңҚеҠЎеҷЁжӯЈзЎ®жҸҗдҫӣдәҶжӣҙж–°ж–Үд»¶пјҲеҰӮ latest.ymlпјүе№¶дё”е…¶ MIME зұ»еһӢи®ҫзҪ®жӯЈзЎ®гҖӮдҫӢеҰӮпјҢеҜ№дәҺ .yml ж–Үд»¶пјҢжңҚеҠЎеҷЁеә”иҝ”еӣһ text/plain жҲ– application/x-yamlпјҢеҗҰеҲҷе®ўжҲ·з«ҜеҸҜиғҪж— жі•и§ЈжһҗгҖӮ
  3. жқғйҷҗй—®йўҳпјҡеңЁ macOS е’Ң Linux дёҠпјҢеә”з”Ёзӣ®еҪ•еҸҜиғҪжІЎжңүеҶҷжқғйҷҗгҖӮжӣҙж–°ж–Үд»¶йҖҡеёёеә”е…ҲдёӢиҪҪеҲ°дёҙж—¶зӣ®еҪ•пјҲеҰӮ app.getPath('temp') жҲ– app.getPath('userData')пјүпјҢйӘҢиҜҒеҗҺеҶҚиҝӣиЎҢжӣҝжҚўж“ҚдҪңгҖӮ
  4. зүҲжң¬з®ЎзҗҶпјҡдҪҝз”ЁиҜӯд№үеҢ–зүҲжң¬пјҲSemantic VersioningпјүпјҢе№¶зЎ®дҝқжү“еҢ…й…ҚзҪ®е’ҢжңҚеҠЎеҷЁдёҠзҡ„зүҲжң¬дҝЎжҒҜеҮҶзЎ®ж— иҜҜгҖӮ
  5. жөӢиҜ•пјҡеҠЎеҝ…еңЁжүҖжңүзӣ®ж Үе№іеҸ°дёҠе……еҲҶжөӢиҜ•жӣҙж–°жөҒзЁӢпјҢеҢ…жӢ¬д»Һж—§зүҲжң¬еҚҮзә§еҲ°ж–°зүҲжң¬гҖҒзҪ‘з»ңдёӯж–ӯгҖҒжӣҙж–°еӨұиҙҘеӣһж»ҡзӯүеңәжҷҜгҖӮ
  6. й”ҷиҜҜеӨ„зҗҶпјҡеҒҡеҘҪзҪ‘з»ңй”ҷиҜҜгҖҒжңҚеҠЎеҷЁй”ҷиҜҜгҖҒж–Үд»¶ж ЎйӘҢеӨұиҙҘзӯүжғ…еҶөзҡ„еӨ„зҗҶе’Ңз”ЁжҲ·жҸҗзӨәгҖӮ
  7. еўһйҮҸжӣҙж–°пјҡelectron-updater ж”ҜжҢҒз”ҹжҲҗе·®еҲҶжӣҙж–°еҢ…пјҢеҸҜд»Ҙжҳҫи‘—еҮҸе°ҸдёӢиҪҪдҪ“з§ҜгҖӮзЎ®дҝқеңЁ electron-builder й…ҚзҪ®дёӯеҗҜз”Ёзӣёе…іеҠҹиғҪгҖӮ
  8. macOS жІҷзӣ’йҷҗеҲ¶пјҡеҰӮжһңеә”з”ЁеҗҜз”ЁдәҶжІҷзӣ’пјҲSandboxпјүпјҢеҸҜиғҪдјҡеҜ№жӣҙж–°ж“ҚдҪңдә§з”ҹйҷҗеҲ¶пјҢйңҖиҰҒд»”з»ҶжөӢиҜ•гҖӮ

рҹ’Һ жҖ»з»“ вҖӢ

йҖүжӢ©е“Әз§Қж–№жЎҲдё»иҰҒеҸ–еҶідәҺдҪ зҡ„е…·дҪ“йңҖжұӮпјҡ

  • иҝҪжұӮеҠҹиғҪе®Ңе–„гҖҒзңҒеҝғеҸҜйқ пјҡзӣҙжҺҘдҪҝз”Ё electron-updater (й…ҚеҗҲ electron-builder)гҖӮ
  • еә”з”ЁејҖжәҗдё”еҸ‘еёғеңЁ GitHubпјҡеҸҜд»Ҙе°қиҜ• update-electron-appгҖӮ
  • йңҖиҰҒжһҒеәҰе®ҡеҲ¶еҢ–зҡ„жӣҙж–°жңҚеҠЎеҷЁйҖ»иҫ‘пјҡз ”з©¶еҶ…зҪ®зҡ„ autoUpdater е№¶иҮӘе»әжңҚеҠЎеҷЁгҖӮ
  • д»…жӣҙж–°жёІжҹ“иҝӣзЁӢиө„жәҗдё”еёҢжңӣиҪ»йҮҸпјҡеҸҜд»ҘиҖғиҷ‘жүӢеҠЁзғӯжӣҙж–°пјҢдҪҶеҠЎеҝ…жіЁж„Ҹе®үе…ЁжҖ§гҖӮ
